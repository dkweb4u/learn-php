Curl Operations

 Without Token
==================
$ch = curl_init();

curl_setopt($ch, CURLOPT_URL, "https://randomuser.me/api");
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$responce = curl_exec($ch);

curl_close($ch);
    
 Without Token
==================
$ch = curl_init();

curl_setopt_array($ch,
    [
      CURLOPT_URL => "url",
      CURLOPT_RETURNTRANSFER => true ,      
      CURLOPT_HTTPHEADER => [ "Authorization: Api-Key token=tokenvalue"],
    ]
);

$responce = curl_exec($ch);

curl_close($ch);
